                + BOOKING_CAR_ID + " INTEGER, "
                + BOOKING_START + " TEXT, "

    }

    // Triggered when DATABASE_VERSION is increased [cite: 110, 111, 124, 126]

        db.execSQL("DROP TABLE IF EXISTS " + TABLE_USERS);
        db.execSQL("DROP TABLE IF EXISTS " + TABLE_BOOKINGS);
        onCreate(db); // [cite: 67]
    }
